Caclulate descriptive statistics on a data frame

d <- data.frame(Normal = x, Uniform = y)  %>% gather(Dist, Value,1:2) 
  
d %>% group_by(Dist) %>% 
  summarise(
            N = n(),
            Avg = mean(Value), 
            SD = sd(Value),
            min = min(Value),
            max = max(Value),
            Var = var(Value)
            )

ANOVA

one way ANOVA

d2 <- data.frame(Group1 = c(7,6,8,6),
                 Group2 = c(9,9,7,8),
                 Group3 = c(3,5,6,5)) %>% gather(Group, Value, 1:3)

res <- aov(Value ~ Group, data = d2)
